// Brute force

class Solution {
public:
    int maxProduct(vector<int>& nums) {
        // brute-force
        int n=nums.size();
        int maxPro=INT_MIN;
        int  pro=1;

        for(int i=0;i<n;i++){
            int curr=nums[i];
            maxPro=max(curr,maxPro);
            for(int j=i;j<n;j++){
               
                pro*=nums[j];
                maxPro=max(pro,maxPro);
            }
            pro=1;
        }
        return maxPro;
    }
};
